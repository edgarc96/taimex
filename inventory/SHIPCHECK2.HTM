CONFIRM PACKING LIST

PicLan-IP/BASIC ^ ^
*
 PL_GET_COOKIE INITIALS FROM 'INIT' ELSE
    ERR = '' ; INITIALS=  ''
    CALL PLW.PAGE('LOGINWH.HTM','',ERR)
    RETURN
 END
*
OPEN 'SHIPMENTS' TO SHIPF ELSE RETURN
PL_GETVAR INVNUM FROM 'INVNUM' ELSE INVNUM = ''
PL_GETVAR MENU FROM 'MENU' ELSE MENU = ''
*
MSG = ''
BEGIN CASE
      CASE INVNUM NE ''
           SHIPNO = FIELD(INVNUM,'-',1)
           IF SHIPNO[1,3] = '11K' THEN SHIPNO = SHIPNO[4,99]
           READ SHIP FROM SHIPF,SHIPNO THEN
              PL_PUTVAR SHIPNO IN 'SHIPNO' ELSE NULL
              PL_PUTVAR INVNUM IN 'INVNUM' ELSE NULL
              ERR = ''
              CALL PLW.PAGE('SHIPCHECK3.HTM','',ERR)
              RETURN
           END ELSE
              INVNUM = ''
              MSG = 'Packing List not in file.'
           END
      CASE MENU NE ''
           ERR = ''
           CALL PLW.PAGE('NBMM.HTM','',ERR)
           RETURN
END CASE
*
PL_PUTVAR INVNUM IN 'INVNUM' ELSE NULL
PL_PUTVAR MSG IN 'MSG' ELSE NULL
