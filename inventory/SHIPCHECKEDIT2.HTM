MSG = ''
PL_GETVAR MENU FROM 'MENU' ELSE MENU = ''

SHIPNO = FIELD(INVNUM,'-',1)
IF SHIPNO[1,3] = '11K' THEN SHIPNO = SHIPNO[4,99]
IF PARTNO[1,2] = '1P' THEN PARTNO = PARTNO[3,99]
PREC = OCONV(PARTNO,'TPARTS;X;;0')
IF PREC = '' THEN
  IF PARTNO[1,1] = 'P' THEN
     PARTNO = PARTNO[2,99]
  END
END

BEGIN CASE
      CASE PARTNO NE ''
           READ SHIP FROM SHIPF,SHIPNO ELSE

              INVNUM = ''
              MSG = 'Cancel Process and start again.'
              PARTNO = ''
           END
           LOCATE(PARTNO,SHIP<13>,1;HIT) THEN
              READ SHIPSCAN FROM SHIPSCANF,SHIPNO ELSE SHIPSCAN = ''
              LINO = SHIP<12,HIT>
           10 LOCATE(LINO,SHIPSCAN<1>,1;HH) ELSE
                 SHIPSCAN<1,-1> = LINO
                 GOTO 10
              END
              SHIPSCAN<2,HH> = 'X'
              SHIPSCAN<3,HH> = SHIPSCAN<3,HH> + 1  ; * LINO BOXES
              SHIPSCAN<4>     = SHIPSCAN<4> + 1    ; * TOTAL PACKLIST BOXES
              SHIPSCAN<5,HH>  = DATE():'*':INITIALS:'*':TIME()
              PARTNO = ''
              WRITE SHIPSCAN ON SHIPSCANF,SHIPNO
              MSG = 'GOOD SCAN!'
              PL_PUTVAR MSG IN 'MSG' ELSE NULL

           END ELSE
              READ SOREC FROM SOF,SHIP<2> THEN
                 LOCATE(PARTNO[2,99],SOREC<57>,1;HK) THEN
                     READ SHIPSCAN FROM SHIPSCANF,SHIPNO ELSE SHIPSCAN = ''
*                     LINO = SHIP<12,HK>
                     LINO = SOREC<31,HK>
                     15 LOCATE(LINO,SHIPSCAN<1>,1;HH) ELSE
                         SHIPSCAN<1,-1> = LINO
                         GOTO 15
                        END
                        SHIPSCAN<2,HH> = 'X'
                        SHIPSCAN<3,HH> = SHIPSCAN<3,HH> + 1  ; * LINO BOXES
                        SHIPSCAN<4>     = SHIPSCAN<4> + 1    ; * TOTAL PACKLIST BOXES
                        SHIPSCAN<5,HH>  = DATE():'*':INITIALS:'*':TIME()
                        PARTNO = ''
                        WRITE SHIPSCAN ON SHIPSCANF,SHIPNO
                        MSG = 'GOOD SCAN!'
                        PL_PUTVAR MSG IN 'MSG' ELSE NULL

                  END ELSE
                        MSG = 'Part No. not in Packing List.'
                        PL_PUTVAR MSG IN 'MSG' ELSE NULL
                  END
              END
           END
      CASE DONE NE ''
          ERR = ''
          PL_PUTVAR INVNUM IN 'INVNUM' ELSE NULL
          CLOSE SHIPF
          CLOSE SOF
          CLOSE PARTSF
          CLOSE SHIPSCANF
          CALL PLW.PAGE('SHIPCHECK3.HTM','',ERR)
          RETURN
      CASE BACK NE ''
           DELETE SHIPSCANF,SHIPNO
           INVNUM = ''
           PL_PUTVAR INVNUM IN 'INVNUM' ELSE NULL
           ERR = ''
           CLOSE SHIPF
           CLOSE SOF
           CLOSE PARTSF
           CLOSE SHIPSCANF
           CALL PLW.PAGE('SHIPCHECKEDIT.HTM','',ERR)
           RETURN
      CASE MENU NE ''
           ERR = ''
           CLOSE SHIPF
           CLOSE SOF
           CLOSE PARTSF
           CLOSE SHIPSCANF
           CALL PLW.PAGE('NBMMEDIT.HTM','',ERR)
           RETURN
END CASE
