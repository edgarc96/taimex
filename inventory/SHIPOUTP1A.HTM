PL_GET_COOKIE INITIALS FROM 'INIT' ELSE
    ERR = '' ; INITIALS=  ''
    CALL PLW.PAGE('LOGINWH.HTM','',ERR)
    RETURN
END
VWHSE.OUT=''

*
OPEN '',"SHIP" TO SHIPF ELSE RETURN
OPEN '',"INV" TO INVF ELSE RETURN
OPEN '',"INVT" TO INVTF ELSE RETURN
OPEN '',"PALMSHIP" TO PALMSHIPF ELSE RETURN
OPEN '',"PALMCHECK" TO PALMCHECKF ELSE RETURN
*
PL_GETVAR INVNUM FROM 'INVNUM' ELSE INVNUM = ''
PL_GETVAR LISEL FROM 'LISEL' ELSE LISEL = ''
PL_GETVAR LINE FROM 'LINE' ELSE LINE = 1
PL_GETVAR LOCCHK FROM 'LOCCHK' ELSE LOCCHK = ''
PL_GETVAR CANCEL7 FROM 'CANCEL7' ELSE CANCEL7 = ''
PL_GETVAR SKIP FROM 'SKIP' ELSE SKIP = ''
PL_GETVAR COUNTER FROM 'COUNTER' ELSE COUNTER = '1'
PL_GETVAR L.ID FROM 'LID' ELSE L.ID = ''
*
IF INVNUM NE '' THEN INVNUM = INVNUM[1,6]
*
ID = INVNUM:"*":LISEL
IF L.ID NE '' THEN
    ID = 'L*':L.ID
END ELSE
    READ PCHK FROM PALMCHECKF, ID THEN
        MSG = 'THIS PACKING LIST NUMBER AND LINE ITEM HAVE ALREADY BEEN PROCESSED. PLEASE CHECK WITH YOUR SUPERVISOR. ':ID
        PL_PUTVAR MSG IN 'MSG' ELSE NULL
        ERR = ''
        CALL PLW.PAGE('PALMERROR3.HTM','',ERR)
        RETURN
    END
END
*
IF SKIP NE "" THEN
    COUNTER = COUNTER + 1
END
IF CANCEL7 NE "" THEN
    READ TMP FROM PALMSHIPF, ID ELSE TMP = ''
    PARTNUM = TMP<7>
    READ INV FROM INVTF,PARTNUM ELSE INV = ""
    IF INV NE "" THEN WRITE INV ON INVF,PARTNUM
    DELETE INVTF,PARTNUM
    DELETE PALMSHIPF, ID
    INVNUM = '' ; LISEL = ''
    PL_PUTVAR INVNUM IN 'INVNUM' ELSE NULL
    PL_PUTVAR LISEL IN 'LISEL' ELSE NULL
    ERR = ''
    CALL PLW.PAGE('STOCKMENU.HTM','',ERR)
    RETURN
END
IF LINE GE 2 THEN
    READ TMP FROM PALMSHIPF, ID ELSE TMP = ''
    TBAL = TMP<4>
END ELSE
    READ SHIP FROM SHIPF,INVNUM ELSE SHIP = ''
    TBAL = SHIP<14,LISEL>
END  
*
READ SHIP FROM SHIPF,INVNUM ELSE SHIP = ''
SONUM = SHIP<2>
PARTNO = SHIP<13,LISEL>
VWHSE = SHIP<45,LISEL>
IF L.ID NE '' THEN
    * Liberty VWHSE
    VWHSE = '000'
    INVNUM = 'L':FIELD(L.ID,'*',1)
    LISEL = FIELD(L.ID,'*',2)
    INV=''
    OPEN 'L.DSLABEL' TO LDS.FILE THEN
        READ L.REC FROM LDS.FILE, L.ID THEN
            PARTNO = L.REC<2>
            IF LINE = 1 THEN
                TBAL = L.REC<7>
            END
        END
    END
END ELSE
    READ INV FROM INVF,PARTNO ELSE INV = ''
END
LOC = ''
LOC.QTY = ''

SHIPQTY = SHIP<14,LISEL>
SHIPQTY1 = OCONV(SHIPQTY,'MR,')


IF VWHSE = '000' THEN
    * output message to user
    OPEN 'VWHSE,' TO VWHSE.FILE THEN
        READ VWHSE.RECORD FROM VWHSE.FILE, VWHSE THEN
            VDESC = OCONV(VWHSE.RECORD<2>,'TCUSTOMER;X;;1')
            IF VDESC = '' THEN
                VDESC = VWHSE.RECORD<1>
            END
            VWHSE.OUT='
&n':'bsp;VWSE: ':VWHSE:' ':VDESC:''
        END
    END
    OPEN 'VWHSEINV,':VWHSE TO VWHSEINV.FILE THEN
        READ VWHSEINV.REC FROM VWHSEINV.FILE, PARTNO THEN
            LOC.COUNT = DCOUNT(VWHSEINV.REC<2>, @VM)
            IF LOC.COUNT > 0 THEN
                * Sort by datecode - oldest first
                S=''    ; * sorted keys
                S.C=1   ; * sorted count
                FOR Y = 1 TO LOC.COUNT
                    * 86C1B1234 [6,4] 9
                    * 88A1234 [4,4] 7
                    DC = VWHSEINV.REC<2,Y>[LEN(VWHSEINV.REC<2,Y>)-3,4]
                    FOR Z = 1 TO S.C+1
                        IF S = '' OR FIELD(S, '*', 1) > DC THEN
                            * shift everything down
                            FOR ZZ = S.C TO Z STEP -1
                                S = S
                            NEXT ZZ
                            * insert key at correct place
                            S = DC:'*':Y
                            * increase sorted key array size
                            S.C+=1
                            GOTO done_sort:
                        END
                    NEXT Z
                    done_sort:
                NEXT Y
                FOR ZZ = COUNTER TO S.C
                    IF VWHSEINV.REC<3,ZZ> > 0 THEN
                        LOC = VWHSEINV.REC<2,ZZ>
                        LOC.QTY = VWHSEINV.REC<3,ZZ>
                        *VWHSE.OUT=LOC
                        GOTO found_vwhse_loc:
                    END
                NEXT ZZ
            END            
        
        END
    END
END
* run normal search for location
CALL SORTINVLOC(PARTNO,INV)
FOR ZZ = COUNTER TO DCOUNT(INV<31>,VM)
    IF INV<32,ZZ> > "0" THEN
        LOC = INV<31,ZZ>
        LOC.QTY = INV<32,ZZ>
        ZZ = DCOUNT(INV<31>,VM)
    END
NEXT ZZ
*
found_vwhse_loc:
*
IF LOCCHK NE "" THEN
    XX = LEN(LOC)
    XX = XX - 4
    CLOC = LOC[1,XX]
    IF LOCCHK NE CLOC THEN
        MSG = 'PART NUMBER SCANNED ':LOCCHK:' DOES NOT MATCH LOC REQUIRED ':CLOC
        PL_PUTVAR MSG IN 'MSG' ELSE NULL
        ERR = ''
        CALL PLW.PAGE('PALMERROR2.HTM','',ERR)
        RETURN
    END
    IF L.ID NE '' THEN
        INVNUM=''
        LISEL=''
    END
    PL_PUTVAR INVNUM IN 'INVNUM' ELSE NULL
    PL_PUTVAR LISEL IN 'LISEL' ELSE NULL
    PL_PUTVAR LINE IN 'LINE' ELSE NULL
    PL_PUTVAR LOC IN 'LOC' ELSE NULL
    PL_PUTVAR LOC.QTY IN 'LOCQTY' ELSE NULL
    PL_PUTVAR COUNTER IN 'COUNTER' ELSE NULL
    PL_PUTVAR L.ID IN 'LID' ELSE NULL
    READ TMP FROM PALMSHIPF,ID ELSE TMP = ''
    TMP<9> = COUNTER
    IF VWHSE NE '' THEN
        TMP<20> = VWHSE
    END
    WRITE TMP ON PALMSHIPF,ID
    ERR = ''
    CALL PLW.PAGE('SHIPOUTP2.HTM','',ERR)
    RETURN
END
