Print Inner Box Label

Scan or enter the part number to generate a label

Instructions:
Scan the part number and click OK to generate the inner box label for this part. 
PicLan-IP/BASIC ^ ^
*
* Initialize variables
PARTNUM = ''
CANCEL2 = ''
ERR = ''
INITIALS = ''
PREC = ''
*
 PL_GET_COOKIE INITIALS FROM 'INIT' ELSE
    ERR = '' ; INITIALS=  ''
    CALL PLW.PAGE('LOGINWH.HTM','',ERR)
    RETURN
 END
* 
PL_GETVAR PARTNUM FROM 'PARTNUM' ELSE PARTNUM = ''
PL_GETVAR CANCEL2 FROM 'CANCEL2' ELSE CANCEL2 = ''
*
IF CANCEL2 NE "" THEN
   ERR = ''
   CALL PLW.PAGE('STOCKMENU.HTM','',ERR)
   RETURN
END
*
IF PARTNUM NE "" THEN
   PREC = OCONV(PARTNUM,'TPARTS;X;;0')
   IF PREC = '' AND PARTNUM[1,1] = 'P' THEN PARTNUM = PARTNUM[2,99]
   PL_PUTVAR PARTNUM IN 'PARTNUM' ELSE NULL
   ERR = ''
   CALL PLW.PAGE('SHIPLCLABELP2EDIT.HTM','',ERR)
   RETURN
END
