 END
 OPEN 'SHIPMENTS' TO SHIPF ELSE
    CLOSE SOUNDSF
    CLOSE INVF
    RETURN
 END

PL_GETVAR SHIPNO FROM 'SHIPNO' ELSE SHIPNO = ''
PL_GETVAR PARTNO FROM 'PARTNO' ELSE PARTNO = ''
PL_GETVAR MSG FROM 'MSG' ELSE MSG = ''
PL_GETVAR SELECT FROM 'SELECT' ELSE SELECT = ''
PL_GETVAR NEWPART FROM 'NEWPART' ELSE NEWPART = ''
PL_GETVAR LINO FROM 'LINO' ELSE LINO = ''
PL_GETVAR CANCEL FROM 'CANCEL' ELSE CANCEL = ''
*
READ SHIP FROM SHIPF,SHIPNO ELSE SHIP = ''
QTY = SHIP<14,LINO>
PARTNO_ORIG = PARTNO
PARTNO = OCONV(PARTNO,'MCU')
READ INV FROM INVF,PARTNO ELSE INV = ''
BINLOC = INV<8,1>
*
PRINT '===== DEBUG START ====='
PRINT 'NEWPART RAW = ':NEWPART
PRINT 'PARTNO RAW  = ':PARTNO
PRINT 'NEWPART LENGTH = ':LEN(NEWPART)
*
* Normalize NEWPART for comparison
NEWPART_CLEAN = NEWPART
* Remove leading '1P' prefix if present
IF NEWPART_CLEAN[1,2]= '1P' THEN NEWPART_CLEAN = NEWPART_CLEAN[3,99]
* Remove leading 'P' prefix if present
IF NEWPART_CLEAN[1,1] = 'P' THEN NEWPART_CLEAN = NEWPART_CLEAN[2,99]
*
* Normalize PARTNO for comparison
PARTNO_CLEAN = PARTNO
IF PARTNO_CLEAN[1,2]= '1P' THEN PARTNO_CLEAN = PARTNO_CLEAN[3,99]
IF PARTNO_CLEAN[1,1] = 'P' THEN PARTNO_CLEAN = PARTNO_CLEAN[2,99]
*
PRINT 'NEWPART_CLEAN = ':NEWPART_CLEAN
PRINT 'PARTNO_CLEAN  = ':PARTNO_CLEAN
PRINT 'MATCH = ':(OCONV(NEWPART_CLEAN,'MCU') = OCONV(PARTNO_CLEAN,'MCU'))
PRINT '===== DEBUG END ====='
SOUNDWAV = ''
BEGIN CASE
      CASE CANCEL NE ''
           PRINT 'CANCEL BUTTON PRESSED - GOING BACK'
           ERR = ''
           CLOSE SOUNDSF
           CLOSE INVF
           CLOSE SHIPF
           CALL PLW.PAGE('STOCKCHECKEDIT.HTM','',ERR)
           RETURN
      CASE NEWPART NE ''
           PRINT 'ENTERING COMPARISON BLOCK'
           IF OCONV(NEWPART_CLEAN,'MCU') NE OCONV(PARTNO_CLEAN,'MCU') THEN
             NEWPART = ''
             MSG = 'Invalid Part No.'
             PRINT 'RESULT: INVALID'
            END ELSE
             MSG = 'VERIFIED'
             NEWPART = ''
             PRINT 'RESULT: VERIFIED'
            END
           PL_PUTVAR MSG IN 'MSG' ELSE NULL
      CASE 1
           PRINT 'NEWPART IS EMPTY - NO COMPARISON'
END CASE
