 PL_GET_COOKIE INITIALS FROM 'INIT' ELSE
    ERR = '' ; INITIALS=  ''
    CALL PLW.PAGE('LOGINWH.HTM','',ERR)
    RETURN
 END
*
OPEN '',"SHIP" TO SHIPF ELSE RETURN
OPEN '',"PALMSHIP" TO PF ELSE RETURN
*
PL_GETVAR INVNUM FROM 'INVNUM' ELSE INVNUM = ''
PL_GETVAR LISEL FROM 'LISEL' ELSE LISEL = ''
PL_GETVAR CANCEL FROM 'CANCEL' ELSE CANCEL = ''
*
IF CANCEL NE "" THEN
   ERR = ''
   CALL PLW.PAGE('STOCKMENU.HTM','',ERR)
   RETURN
END
*
IF INVNUM NE '' THEN INVNUM = INVNUM[1,6]

IF INVNUM NE '' THEN
  TT1 = "Line#"
END ELSE
  TT1 = ''
END
*
LISEL=''
IF INVNUM NE "" THEN
        READ SHIP FROM SHIPF,INVNUM ELSE SHIP = ''
        IF SHIP = '' THEN 
            * disable liberty drop ships 
            MSG = 'Invalid Shipment No.'
            PL_PUTVAR MSG IN 'MSG' ELSE NULL
            ERR = ''
            CALL PLW.PAGE('PALMERROR2.HTM','',ERR)        
        END 
        * liberty drop ships - no longer used 2019
        IF SHIP = '' THEN
            * Look for liberty shipment
            EXECUTE 'END'
            EXECUTE 'SELECT L.DSLABEL WITH 6 "':INVNUM:'"'
            L.ID = ''
            IF SYSTEM(11) > 0 THEN
                * ask 
                LOOP
                    READNEXT ID THEN
                        L.ID = ID
                    END ELSE
                        EXIT
                    END
                REPEAT
            END
            IF L.ID = '' THEN
                MSG = 'Invalid Shipment No.'
                PL_PUTVAR MSG IN 'MSG' ELSE NULL
                ERR = ''
                CALL PLW.PAGE('PALMERROR2.HTM','',ERR)
                RETURN
            END ELSE
                *L.ID:='*':INVNUM
                PL_PUTVAR L.ID IN 'LID' ELSE NULL
                ERR = ''
                CALL PLW.PAGE('SHIPOUTLIB1.HTM','',ERR)
                RETURN
            END
        END
   LISEL = ''
   *CALL BUILDLISELECTAUTO2(INVNUM,LISEL)
   * http://cdenb2.cde.com/shipoutp1a.htm?LISEL=1&INVNUM=640057
    *SEL = \SSELECT SHIP WITH 0 "\:INVNUM:\" BY-EXP 12\
    *EXECUTE SEL CAPTURING ERR


    READ DREC FROM SHIPF,INVNUM THEN 
        FOR I = 1 TO DCOUNT(DREC<12>,@VM)
            URL = 'http://cdenb2.cde.com/shipoutp1a.htm?LISEL=':I:'&INVNUM=':INVNUM
            DESC = 'LINE ':I:' ':DREC<13, I>
            LISEL:='':DESC:'<':'br>'
        NEXT I 
    END 

END
*
PL_PUTVAR TT1 IN 'TT1' ELSE NULL
PL_PUTVAR LISEL IN 'LISEL' ELSE NULL
