PL_GETVAR PARTNUM FROM 'PARTNUM' ELSE PARTNUM = ''
PL_GETVAR BACK FROM 'BACK' ELSE BACK = ''
PL_GETVAR CREATE FROM 'CREATE' ELSE CREATE = ''
PL_GETVAR OK FROM 'OK' ELSE OK = ''
PL_GETVAR CLEAR FROM 'CLEAR' ELSE CLEAR = ''
*
RESULT = ''
IQC = ''
IQCDATA = ''
MSG = ''
DESCR = ''
TEST = ''
PREC = ''
*
* Manejar CLEAR primero - resetear todo
IF CLEAR NE '' THEN
  PARTNUM = ''
  MSG = ''
  RESULT = ''
  IQC = ''
  IQCDATA = ''
  CREATE = ''
  BACK = ''
  OK = ''
  * Limpiar variables de sesión
  PL_PUTVAR '' IN 'PARTNUM' ELSE NULL
  PL_PUTVAR '' IN 'CREATE' ELSE NULL
  PL_PUTVAR '' IN 'BACK' ELSE NULL
  PL_PUTVAR '' IN 'OK' ELSE NULL
  PL_PUTVAR '' IN 'CLEAR' ELSE NULL
  GOTO 99
END
*
IF BACK NE '' THEN
  PL_PUTVAR PARTNUM IN 'PARTNUM' ELSE NULL
  ERR = ''
  CALL PLW.PAGE('INVIRMEDIT.HTM','',ERR)
  RETURN
END
*
IF CREATE NE '' THEN
  PL_PUTVAR PARTNUM IN 'PARTNUM' ELSE NULL
  ERR = ''
  CALL PLW.PAGE('INVPULL1EDIT.HTM','',ERR)
  RETURN
END
*
* Si hay PARTNUM, generar resultados automáticamente
PARTNUM = TRIM(PARTNUM)
PARTSF.OPEN = 0
INVF.OPEN = 0
IF PARTNUM NE '' THEN
  OPEN 'PARTS' TO PARTSF ELSE
    MSG = 'UNABLE TO OPEN PARTS FILE'
    GOTO 99
  END
  PARTSF.OPEN = 1
  OPEN 'INV' TO INVF ELSE INVF.OPEN = 0
  IF INVF.OPEN = 0 THEN INVF.OPEN = 0 ELSE INVF.OPEN = 1
*
  PREC = OCONV(PARTNUM,'TPARTS;X;;0')
  IF PREC = '' AND PARTNUM[1,1] = 'P' THEN
    PARTNUM = PARTNUM[2,99]
  END
  PARTNUM = TRIM(PARTNUM)
*
  READ TEST FROM PARTSF,PARTNUM ELSE
    MSG = 'INVALID PART#'
    CLOSE PARTSF
    IF INVF.OPEN = 1 THEN CLOSE INVF
    GOTO 99
  END
*
  * Verificar si hay datos de inventario
  IF INVF.OPEN = 1 THEN
    READ INV FROM INVF,PARTNUM ELSE INV = ''
  END ELSE
    INV = ''
  END
  BOH = INV<1>
  IF BOH = '' OR BOH = 0 THEN
    * No hay inventario, dejar RESULT vacío para mostrar "No Results"
    MSG = ''
    RESULT = ''
    IQC = ''
    IQCDATA = ''
  END ELSE
    * Hay inventario, generar resultados
    MSG = ''
    DESCR = TEST<2>
    IF DESCR = '' THEN DESCR = 'N/A'
    RESULT = 'Part Number: ' : PARTNUM
    RESULT = RESULT : CHAR(10) : 'Description: ' : DESCR
    RESULT = RESULT : CHAR(10) : 'Status: Active'
    IQC = 'Incoming Inspection'
    IQCDATA = 'No open receipts at this time'
  END
*
  * Cerrar archivos SIEMPRE
  CLOSE PARTSF
  IF INVF.OPEN = 1 THEN CLOSE INVF
END
*
99 CONTINUE

  RETURN
