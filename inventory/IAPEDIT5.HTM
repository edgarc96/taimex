PL_GETVAR YCODE FROM 'YCODE' ELSE YCODE = ''
PL_GETVAR MCODE FROM 'MCODE' ELSE MCODE = ''
PL_GETVAR QTY FROM 'QTY' ELSE QTY = ''
PL_GETVAR LOCTN FROM 'LOCTN' ELSE LOCTN = ''
PL_GETVAR PARTNO FROM 'PARTNO' ELSE PARTNO = ''
PL_GETVAR MENU FROM 'MENU' ELSE MENU = ''
PL_GETVAR NEWLOC FROM 'NEWLOC' ELSE NEWLOC = ''
PL_GETVAR OK FROM 'OK' ELSE OK = ''
PL_GETVAR BOXNUM FROM 'BOXNUM' ELSE BOXNUM = ''

OPEN '','INV' TO INVF ELSE RETURN
OPEN '','TRACK.REC' TO TRACKF ELSE
   CLOSE INVF
   RETURN
END
ERR = ''
MSG = ''
VWHSEMESSAGE = ''
*
IF VWHSE NE '' THEN
    OPEN 'VWHSE' TO VWHSE.FILE THEN
        READ VWHSE.RECORD FROM VWHSE.FILE, VWHSE THEN
            VWHSEMESSAGE = 'VWHSE - ':VWHSE:' ':OCONV(VWHSE.RECORD<2>,'TCUSTOMER;X;;1')
        END
        CLOSE VWHSE.FILE
    END
END
PL_PUTVAR VWHSEMESSAGE IN 'VWHSEMESSAGE' ELSE NULL
*
IF QTY[1,1] = 'Q' THEN QTY = QTY[2,99]
IF QTY NE '' AND NOT(NUM(QTY)) THEN
  MSG = 'QTY MUST BE NUMERIC'
  PL_PUTVAR MSG IN 'MSG' ELSE NULL
  ERR = ''
  CLOSE INVF
  CLOSE TRACKF
  CALL PLW.PAGE('PALMERROREDIT2.HTM','',ERR)
  RETURN
  QTY = ''
END
*
IF MENU NE '' THEN
   ERR = ''
   CLOSE INVF
   CLOSE TRACKF
   CALL PLW.PAGE('STOCKCHECK.HTM','',ERR)
   RETURN
END
*
IF NEWLOC NE '' THEN
   ERR = ''
   CLOSE INVF
   CLOSE TRACKF
   CALL PLW.PAGE('IAPEDIT1.HTM','',ERR)
   RETURN
END
*
IF OK NE '' AND QTY NE '' AND MSG = '' THEN
    YRMO = YCODE:MCODE
    NEWLOCTN = LOCTN:YRMO
    IF VWHSE NE '' THEN
        ERR = ''
        IF QTY GE 0 THEN
            CALL PUTVWHSELOC(VWHSE, PARTNO, QTY, LOCTN, YCODE:MCODE, ERR)
        END ELSE
            CALL TAKEVWHSELOC(VWHSE, PARTNO, (QTY*-1), NEWLOCTN, ERR)
        END
        IF ERR NE '' THEN
            MSG = ERR
            ERR = ''
            PL_PUTVAR MSG IN 'MSG' ELSE NULL
            CLOSE INVF
            CLOSE TRACKF
            CALL PLW.PAGE('PALMERROREDIT2.HTM','',ERR)
            RETURN
        END
    END ELSE
        READ INV FROM INVF, PARTNO ELSE INV = ''
100 *
        LOCATE(NEWLOCTN,INV<31>,1;HIT) ELSE
            INV<31,-1> = NEWLOCTN
            GOTO 100
        END
        INV<32,HIT> += QTY
        WRITE INV ON INVF, PARTNO
    END
    NEWID = PARTNO:'*':NEWLOCTN:'*':QTY:'*':OCONV(DATE(),'D2-'):'*':OCONV(TIME(),'MTS'):'*':'SYS'
    WRITE '' ON TRACKF, NEWID
    LOCTOT = LOCTOT + QTY
    PARTNO = ''
    QTY = ''
    BOXNUM = BOXNUM + 1
    PL_PUTVAR PARTNO IN 'PARTNO' ELSE NULL
    PL_PUTVAR QTY IN 'QTY' ELSE NULL
    PL_PUTVAR LOCTN IN 'LOCTN' ELSE NULL
    PL_PUTVAR BOXNUM IN 'BOXNUM' ELSE NULL
    PL_PUTVAR LOCTOT IN 'LOCTOT' ELSE NULL
    PL_PUTVAR VWHSE IN 'VWHSE' ELSE NULL
    PL_PUTVAR YCODE IN 'YCODE' ELSE NULL
    PL_PUTVAR MCODE IN 'MCODE' ELSE NULL
    ERR = ''
    CLOSE INVF
    CLOSE TRACKF
    CALL PLW.PAGE('IAPEDIT4.HTM','',ERR)
    RETURN
END
