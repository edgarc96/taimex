PL_GET_COOKIE INITIALS FROM 'INIT' ELSE
    ERR = '' ; INITIALS=  ''
    CALL PLW.PAGE('LOGINWH.HTM','',ERR)
    RETURN
END

L.DETAILS=''
OPEN '',"PALMSHIP" TO PF ELSE RETURN
OPEN 'L.DSLABEL' TO LDS.FILE ELSE RETURN
*
PL_GETVAR L.ID FROM 'LID' ELSE L.ID = ''
PL_GETVAR CANCEL FROM 'LCANCEL' ELSE CANCEL = ''
PL_GETVAR OKAY FROM 'LCONTINUE' ELSE OKAY=''
*
READ L.REC FROM LDS.FILE, L.ID THEN
    BR='<':'br>'
    L.DETAILS:= L.REC<9>:BR
    L.DETAILS:= 'Order #':FIELD(L.ID,'*',1):' LI #':FIELD(L.ID,'*',2):BR
    L.DETAILS:= 'Part No. ': L.REC<2>:' QTY ':L.REC<7>:BR

END
IF CANCEL NE "" THEN
   ERR = ''
   CALL PLW.PAGE('NBMMEDIT.HTM','',ERR)
   RETURN
END
*
IF OKAY NE '' THEN
    PL_PUTVAR L.ID IN 'LID' ELSE NULL
    * new IDs for Liberty shipments
    DELETE PF,'L*':L.ID
    EXECUTE "SP-ASSIGN F4"
    PRINTER ON
    CALL PRINTLIBDSPICKSLIP(L.ID)
    PRINTER OFF    
    EXECUTE "SP-ASSIGN F0"
    ERR = ''
    CALL PLW.PAGE('SHIPOUTP1A.HTM','',ERR)
    RETURN
END
